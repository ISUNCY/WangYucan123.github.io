游戏源代码(单击代码右上角^可折叠代码)：

#include <stdio.h>
#include <time.h>
#include <graphics.h>
#include <windows.h>
#include <mmsystem.h>
#include <stdlib.h>
#include <conio.h>
#pragma comment(lib,"Winmm.lib")
#define WIDTH  680    //窗口宽度
#define HEIGHT 680    //窗口高度



IMAGE bk1;                    //开始界面背景图片
IMAGE bk2;                    //游戏中背景图片
MOUSEMSG msg;                //鼠标消息
bool black = true;            //判断棋子是否为黑色
int blackWin;                //判断黑方是否获胜（值为1时黑方胜，为-1时白方胜，为2时为平局，为0时则未分胜负）
int rounds = 0;                //回合数

struct Chess
{
    bool isChess = false;    //判断此处是否有棋子
    int x;                    //棋子的x坐标
    int y;                    //棋子的y坐标
    bool black;                //棋子的颜色是否为黑色
    bool lately = false;    //是否为最近一次下的棋
}GameMap[18][18];


void initdata(void);        //初始化数据
int beginGame(void);        //游戏开始界面
void drawMap(void);            //画出地图
void changeMap(void);        //改变地图（用户落子）
void WinOrLose(void);        //判断胜负

int main(void)
{
    initgraph(WIDTH, HEIGHT);


    BeginBatchDraw();
    initdata();                    //初始化数据
    if (beginGame() == -1)        //开始界面
    {
        closegraph();
        return 0;
    }
    closegraph();
    initgraph(WIDTH + 220, HEIGHT);        //重新创建一个大一点的窗口

    initdata();
    BeginBatchDraw();
    cleardevice();
    mciSendString("play .//src//music//高山流水.mp3 repeat", 0, 0, 0);    //循环播放背景音乐
    HANDLE handle;
    //创建一个线程来画地图
    handle = CreateThread(NULL, 0, (LPTHREAD_START_ROUTINE)drawMap, NULL, NULL, NULL);
    while (1)            //循环改变地图，画地图
    {
        changeMap();
    }

    closegraph();
    return 0;
}

//初始化数据
void initdata(void)
{
    settextcolor(LIGHTGREEN);
    setbkcolor(RGB(230, 230, 230));
    cleardevice();
    setlinecolor(BLACK);
    setbkmode(TRANSPARENT);
    loadimage(&bk1, ".//src//image//bk1.jpg", WIDTH, HEIGHT);
    loadimage(&bk2, ".//src//image//bk2.jpg", WIDTH, HEIGHT);
}

//游戏开始界面
int beginGame(void)
{
    putimage(0, 0, &bk1);
    settextstyle(90, 0, "楷体");
    outtextxy(220, 10, "五子棋");
    settextcolor(BROWN);
    setfillcolor(RGB(255, 255, 0));
    settextstyle(HEIGHT / 5 * 1 / 2, 0, "楷体");
    outtextxy((WIDTH - textwidth("开始")) / 2, (HEIGHT / 5 * 3) - textheight("开始"), "开始");
    outtextxy((WIDTH - textwidth("退出")) / 2, (HEIGHT / 5 * 4) - textheight("退出"), "退出");
    outtextxy((WIDTH - textwidth("关于")) / 2, (HEIGHT / 5 * 5) - textheight("关于"), "关于");
    FlushBatchDraw();
    while (1)
    {
        if (MouseHit())
            msg = GetMouseMsg();
        if (msg.x > (WIDTH - textwidth("开始")) / 2 &&
            msg.y > (HEIGHT / 5 * 3) - textheight("开始") &&
            msg.x < (WIDTH - textwidth("开始")) / 2 + textwidth("开始") &&
            msg.y < (HEIGHT / 5 * 3) - textheight("开始") + textwidth("开始") / 2)
        {
            fillroundrect((WIDTH - textwidth("开始")) / 2, (HEIGHT / 5 * 3) - textheight("开始"),
                (WIDTH - textwidth("开始")) / 2 + textwidth("开始"),
                (HEIGHT / 5 * 3) - textheight("开始") + textwidth("开始") / 2,
                10, 10);
            outtextxy((WIDTH - textwidth("开始")) / 2, (HEIGHT / 5 * 3) - textheight("开始"), "开始");
        }
        if (msg.x >(WIDTH - textwidth("退出")) / 2 &&
            msg.y >(HEIGHT / 5 * 4) - textheight("退出") &&
            msg.x < (WIDTH - textwidth("退出")) / 2 + textwidth("退出") &&
            msg.y < (HEIGHT / 5 * 4) - textheight("退出") + textwidth("退出") / 2)
        {
            fillroundrect((WIDTH - textwidth("退出")) / 2, (HEIGHT / 5 * 4) - textheight("退出"),
                (WIDTH - textwidth("退出")) / 2 + textwidth("退出"),
                (HEIGHT / 5 * 4) - textheight("退出") + textwidth("退出") / 2,
                10, 10);
            outtextxy((WIDTH - textwidth("退出")) / 2, (HEIGHT / 5 * 4) - textheight("退出"), "退出");
        }
        if (msg.x >(WIDTH - textwidth("关于")) / 2 &&
            msg.y >(HEIGHT / 5 * 5) - textheight("关于") &&
            msg.x < (WIDTH - textwidth("关于")) / 2 + textwidth("关于") &&
            msg.y < (HEIGHT / 5 * 5) - textheight("关于") + textwidth("关于") / 2)
        {
            fillroundrect((WIDTH - textwidth("关于")) / 2, (HEIGHT / 5 * 5) - textheight("关于"),
                (WIDTH - textwidth("关于")) / 2 + textwidth("关于"),
                (HEIGHT / 5 * 5) - textheight("关于") + textwidth("关于") / 2,
                10, 10);
            outtextxy((WIDTH - textwidth("关于")) / 2, (HEIGHT / 5 * 5) - textheight("关于"), "关于");
        }
        if (msg.uMsg == WM_LBUTTONUP)
        {
            if (msg.x >(WIDTH - textwidth("开始")) / 2 &&
                msg.y > (HEIGHT / 5 * 3) - textheight("开始") &&
                msg.x < (WIDTH - textwidth("开始")) / 2 + textwidth("开始") &&
                msg.y < (HEIGHT / 5 * 3) - textheight("开始") + textwidth("开始") / 2)
            {
                return 0;
            }
            if (msg.x >(WIDTH - textwidth("退出")) / 2 &&
                msg.y >(HEIGHT / 5 * 4) - textheight("退出") &&
                msg.x < (WIDTH - textwidth("退出")) / 2 + textwidth("退出") &&
                msg.y < (HEIGHT / 5 * 4) - textheight("退出") + textwidth("退出") / 2)
            {
                closegraph();
                return -1;
            }
            if (msg.x >(WIDTH - textwidth("关于")) / 2 &&
                msg.y >(HEIGHT / 5 * 5) - textheight("关于") &&
                msg.x < (WIDTH - textwidth("关于")) / 2 + textwidth("关于") &&
                msg.y < (HEIGHT / 5 * 5) - textheight("关于") + textwidth("关于") / 2)
            {
                cleardevice();
                settextstyle(20, 0, "楷体");
                outtextxy(0, 0, "本应用用c/c++编写,作者:王宇灿");
                FlushBatchDraw();
                Sleep(3000);
            }
        }
        if (!((msg.x >(WIDTH - textwidth("开始")) / 2 &&
            msg.y > (HEIGHT / 5 * 3) - textheight("开始") &&
            msg.x < (WIDTH - textwidth("开始")) / 2 + textwidth("开始") &&
            msg.y < (HEIGHT / 5 * 3) - textheight("开始") + textwidth("开始") / 2) ||
            (msg.x >(WIDTH - textwidth("退出")) / 2 &&
            msg.y >(HEIGHT / 5 * 4) - textheight("退出") &&
            msg.x < (WIDTH - textwidth("退出")) / 2 + textwidth("退出") &&
            msg.y < (HEIGHT / 5 * 4) - textheight("退出") + textwidth("退出") / 2) ||
            (msg.x >(WIDTH - textwidth("关于")) / 2 &&
            msg.y >(HEIGHT / 5 * 5) - textheight("关于") &&
            msg.x < (WIDTH - textwidth("关于")) / 2 + textwidth("关于") &&
            msg.y < (HEIGHT / 5 * 5) - textheight("关于") + textwidth("关于") / 2)))
        {
            putimage(0, 0, &bk1);
            settextcolor(LIGHTGREEN);
            settextstyle(90, 0, "楷体");
            outtextxy(220, 10, "五子棋");
            settextcolor(BROWN);
            setfillcolor(RGB(255, 255, 0));
            settextstyle(HEIGHT / 5 * 1 / 2, 0, "楷体");
            outtextxy((WIDTH - textwidth("开始")) / 2, (HEIGHT / 5 * 3) - textheight("开始"), "开始");
            outtextxy((WIDTH - textwidth("退出")) / 2, (HEIGHT / 5 * 4) - textheight("退出"), "退出");
            outtextxy((WIDTH - textwidth("关于")) / 2, (HEIGHT / 5 * 5) - textheight("关于"), "关于");
        }
        FlushBatchDraw();
    }
}


//画地图
void drawMap(void)
{
    while (1)
    {
        cleardevice();
        putimage(0, 0, &bk2);
        setlinecolor(BLACK);
        for (int i = 0; i < 18; i++)
        {
            for (int j = 0; j <= 18; j++)
            {
                line(i * 40, 0, i * 40, WIDTH);
                line(0, j * 40, HEIGHT, j * 40);
            }
        }
        setfillcolor(RED);
        solidcircle(WIDTH / 2 - 20, HEIGHT / 2 - 20, 3);                        //中间的小黑点
        solidcircle(WIDTH / 2 - 40 * 4 - 20, HEIGHT / 2 - 40 * 4 - 20, 3);        //左上角的小黑点
        solidcircle(WIDTH / 2 + 40 * 4 - 20, HEIGHT / 2 - 40 * 4 - 20, 3);        //右上角的小黑点
        solidcircle(WIDTH / 2 - 40 * 4 - 20, HEIGHT / 2 + 40 * 4 - 20, 3);        //左下角的小黑点
        solidcircle(WIDTH / 2 + 40 * 4 - 20, HEIGHT / 2 + 40 * 4 - 20, 3);        //右下角的小黑点
        solidcircle(WIDTH / 2 - 20, HEIGHT / 2 + 40 * 4 - 20, 3);                //下面的小黑点
        solidcircle(WIDTH / 2 - 20, HEIGHT / 2 - 40 * 4 - 20, 3);                //上面的小黑点
        solidcircle(WIDTH / 2 - 40 * 4 - 20, HEIGHT / 2 - 20, 3);                //左面的小黑点
        solidcircle(WIDTH / 2 + 40 * 4 - 20, HEIGHT / 2 - 20, 3);                //右面的小黑点
        //根据GameMap的值画出棋子
        for (int i = 0; i < 18; i++)
        {
            for (int j = 0; j < 18; j++)
            {
                if (GameMap[i][j].isChess)
                {
                    if (GameMap[i][j].black){
                        setfillcolor(BLACK);
                    }
                    else                    {
                        setfillcolor(WHITE);
                    }
                    solidcircle(GameMap[i][j].x, GameMap[i][j].y, 10);
                    if (GameMap[i][j].lately){
                        setfillcolor(LIGHTRED);
                        solidcircle(GameMap[i][j].x, GameMap[i][j].y, 3);
                        GameMap[i][j].lately = false;
                    }
                }
            }
        }

        WinOrLose();                //判断胜负

        //画出右侧提示框
        setfillcolor(RGB(200, 200, 200));
        solidrectangle(WIDTH, 0, WIDTH + 10, HEIGHT);

        //显示对局结果
        settextstyle(20, 0, "宋体");
        settextcolor(LIGHTGREEN);
        outtextxy(WIDTH + 10, 10, "对局结果：");
        settextstyle(30, 0, "楷体");
        if (blackWin == 1)
        {
            settextcolor(LIGHTBLUE);
            outtextxy(WIDTH + 100, 0, "黑方胜利！");
        }
        else if (blackWin == -1)
        {
            settextcolor(LIGHTRED);
            outtextxy(WIDTH + 100, 0, "白方胜利！");
        }
        else if (blackWin == 2)
        {
            settextcolor(RGB(255, 0, 255));
            outtextxy(WIDTH + 100, 0, "平局！");
            FlushBatchDraw();
            getchar();
        }
        else
        {
            settextcolor(RGB(50, 50, 50));
            outtextxy(WIDTH + 100, 0, "未分胜负");
        }

        //显示执行方
        settextstyle(20, 0, "宋体");
        settextcolor(RGB(185, 122, 87));
        outtextxy(WIDTH + 10, 50, "行棋方：");
        settextstyle(30, 0, "楷体");
        if (black){
            settextcolor(LIGHTBLUE);
            outtextxy(WIDTH + 80, 45, "黑方");
        }
        else
        {
            settextcolor(LIGHTRED);
            outtextxy(WIDTH + 80, 45, "白方");
        }

        //显示回合数
        settextstyle(20, 0, "宋体");
        settextcolor(LIGHTBLUE);
        outtextxy(WIDTH + 10, 90, "回合数：");
        settextstyle(30, 0, NULL);
        settextcolor(RGB(255, 127, 39));            //橘色
        char rounds_s[10];                            //outtextxy不支持打印int类型，所以转换为字符串类型(round_str)
        sprintf_s(rounds_s, "%d", rounds);
        outtextxy(WIDTH + 80, 85, rounds_s);        //itoa()函数功能是：把整形数值转换为字符串，以此实现打印输出

        //显示已用时间
        settextstyle(20, 0, "宋体");
        settextcolor(RGB(120, 50, 140));
        outtextxy(WIDTH + 10, 130, "已用时间：");
        settextstyle(30, 0, "宋体");
        settextcolor(RGB(40, 240, 230));
        static int FirstTime = (int)time(NULL);        //首次记录值
        int nowTime = (int)time(NULL);                //现在的时间
        char time_s[10];                        //已用时间（time_str）
        sprintf_s(time_s, "%d:%d:%d", (nowTime - FirstTime) / 3600 % 60, (nowTime - FirstTime) / 60 % 60, (nowTime - FirstTime) % 60);
        outtextxy(WIDTH + 100, 125, time_s);
        FlushBatchDraw();
        if (blackWin != 0)
            getchar();
        Sleep(50);            //为防止cpu占用率过高和闪屏，小睡一下

    }
}

//改变地图（用户落子）
void changeMap(void)
{
    msg = GetMouseMsg();
    while (msg.uMsg == WM_MOUSEMOVE || msg.uMsg == WM_LBUTTONUP)
    {
        msg = GetMouseMsg();
    }
    if (msg.uMsg == WM_LBUTTONDOWN)
    {
        //通过判断鼠标位置，确定棋子位置
        for (int i = 0; i <= WIDTH; i += 40)
        {
            for (int j = 0; j <= HEIGHT; j += 40)
            {
                if (msg.x > i - 20 && msg.y > j - 20 && msg.x < i + 20 && msg.y < j + 20 && !GameMap[i / 40][j / 40].isChess)
                {
                    GameMap[i / 40][j / 40].isChess = true;
                    GameMap[i / 40][j / 40].x = i;
                    GameMap[i / 40][j / 40].y = j;
                    if (black)
                    {
                        GameMap[i / 40][j / 40].black = true;
                        black = false;
                    }
                    else
                    {
                        GameMap[i / 40][j / 40].black = false;
                        black = true;
                        rounds++;
                    }
                    GameMap[i / 40][j / 40].lately = true;
                }
            }
        }
    }
}

//判断胜负
void WinOrLose(void)
{
    int filled = 0;                        //判断棋盘是否充满
    for (int i = 0; i < 25; i++)
    {
        for (int j = 0; j < 25; j++)
        {
            //横排为5个
            if (GameMap[i][j].isChess && GameMap[i + 1][j].isChess && GameMap[i + 2][j].isChess &&
                GameMap[i + 3][j].isChess && GameMap[i + 4][j].isChess &&
                GameMap[i][j].black == GameMap[i + 1][j].black &&
                GameMap[i][j].black == GameMap[i + 4][j].black &&
                GameMap[i + 2][j].black == GameMap[i + 3][j].black &&
                GameMap[i + 2][j].black == GameMap[i][j].black){
                setlinecolor(LIGHTRED);
                line(GameMap[i][j].x, GameMap[i][j].y, GameMap[i + 4][j].x, GameMap[i + 4][j].y);
                if (GameMap[i][j].black)
                    blackWin = 1;
                else
                    blackWin = -1;
            }
            //竖排有5个
            else if (GameMap[i][j].isChess && GameMap[i][j + 1].isChess && GameMap[i][j + 2].isChess &&
                GameMap[i][j + 3].isChess && GameMap[i][j + 4].isChess &&
                GameMap[i][j].black == GameMap[i][j + 1].black &&
                GameMap[i][j].black == GameMap[i][j + 4].black &&
                GameMap[i][j + 2].black == GameMap[i][j + 3].black &&
                GameMap[i][j + 2].black == GameMap[i][j].black){
                setlinecolor(LIGHTRED);
                line(GameMap[i][j].x, GameMap[i][j].y, GameMap[i][j + 4].x, GameMap[i][j + 4].y);
                if (GameMap[i][j].black)
                    blackWin = 1;
                else
                    blackWin = -1;
            }
            //右斜有5个
            else if (GameMap[i][j].isChess && GameMap[i + 1][j + 1].isChess && GameMap[i + 2][j + 2].isChess &&
                GameMap[i + 3][j + 3].isChess && GameMap[i + 4][j + 4].isChess &&
                GameMap[i][j].black == GameMap[i + 1][j + 1].black &&
                GameMap[i + 1][j + 1].black == GameMap[i + 4][j + 4].black &&
                GameMap[i + 2][j + 2].black == GameMap[i + 3][j + 3].black &&
                GameMap[i + 2][j + 2].black == GameMap[i][j].black){
                setlinecolor(LIGHTRED);
                line(GameMap[i][j].x, GameMap[i][j].y, GameMap[i + 4][j + 4].x, GameMap[i + 4][j + 4].y);
                if (GameMap[i][j].black)
                    blackWin = 1;
                else
                    blackWin = -1;

            }
            //左斜有5个
            else if (GameMap[i][j].isChess && GameMap[i - 1][j + 1].isChess && GameMap[i - 2][j + 2].isChess &&
                GameMap[i - 3][j + 3].isChess && GameMap[i - 4][j + 4].isChess &&
                GameMap[i][j].black == GameMap[i - 1][j + 1].black &&
                GameMap[i - 1][j + 1].black == GameMap[i - 4][j + 4].black &&
                GameMap[i - 2][j + 2].black == GameMap[i - 3][j + 3].black &&
                GameMap[i - 2][j + 2].black == GameMap[i][j].black){
                setlinecolor(LIGHTRED);
                line(GameMap[i][j].x, GameMap[i][j].y, GameMap[i - 4][j + 4].x, GameMap[i - 4][j + 4].y);
                if (GameMap[i][j].black)
                    blackWin = 1;
                else
                    blackWin = -1;
            }
            //判断棋子是否充满棋盘
            if (GameMap[i][j].isChess)
                filled++;
            if (filled == 18 * 18)
                blackWin = 2;
        }
    }
}
